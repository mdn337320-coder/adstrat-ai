import React, { useState } from 'react';
import { GeminiService } from '../services/geminiService';
import { useAuth } from '../hooks/useAuth';

const CreativeStudio: React.FC = () => {
  const { logout, refresh, fetchWithAuth } = useAuth();
  const [productName, setProductName] = useState('');
  const [productDetails, setProductDetails] = useState('');
  const [renderMode, setRenderMode] = useState<'variant' | 'lifestyle'>('variant');
  const [image, setImage] = useState<string | null>(null);
  const [inspirationImage, setInspirationImage] = useState<string | null>(null);
  const [isLoading, setIsLoading] = useState(false);
  const [result, setResult] = useState<{ description: string; images: string[] } | null>(null);
  const [activeJobId, setActiveJobId] = useState<number | null>(null);
  const [jobStatus, setJobStatus] = useState<string | null>(null);
  const [error, setError] = useState<string | null>(null);
  const [hasKey, setHasKey] = useState<boolean | null>(null);
  const [quality, setQuality] = useState<'standard' | 'studio'>('studio');
  const [isRegeneratingCopy, setIsRegeneratingCopy] = useState(false);
  const [isCopied, setIsCopied] = useState(false);
  const [showPromptPreview, setShowPromptPreview] = useState(false);

  React.useEffect(() => {
    const checkKey = async () => {
      // @ts-ignore
      const selected = await window.aistudio?.hasSelectedApiKey();
      setHasKey(!!selected);
    };
    checkKey();
    const interval = setInterval(checkKey, 2000);
    return () => clearInterval(interval);
  }, []);

  const handleOpenKeySelector = async () => {
    // @ts-ignore
    await window.aistudio?.openSelectKey();
  };

  const handleImageUpload = (e: React.ChangeEvent<HTMLInputElement>, type: 'product' | 'inspiration') => {
    const file = e.target.files?.[0];
    if (file) {
      const reader = new FileReader();
      reader.onloadend = () => {
        if (type === 'product') setImage(reader.result as string);
        else setInspirationImage(reader.result as string);
      };
      reader.readAsDataURL(file);
    }
  };

  const handleProcess = async () => {
    if (!image || !productName) return;
    if (quality === 'studio' && !hasKey) {
      handleOpenKeySelector();
      return;
    }
    setIsLoading(true);
    setJobStatus('processing');
    setError(null);
    setResult(null);
    const token = localStorage.getItem('token');
    
    try {
      // 1. Create the job on the server
      const res = await fetchWithAuth('/api/creative/generate', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({ 
          productName, 
          productImage: image, 
          inspirationImage,
          type: renderMode,
          quality
        })
      });
      
      if (!res.ok) return;
      const data = await res.json();
      if (data.error) throw new Error(data.error);
      
      const jobId = data.jobId;
      setActiveJobId(jobId);

      // 2. Process the AI generation on the client
      const service = new GeminiService();
      const generatedImages = await service.generateCreativeStudioImages(image, productName, productDetails, renderMode, inspirationImage || undefined, quality);
      
      if (!generatedImages || generatedImages.length === 0) {
        throw new Error("The neural engine failed to synthesize assets. Please verify your source images.");
      }

      // 3. Generate initial product copy
      const productDescription = await service.generateProductDescription(productName);

      // 4. Save the results back to the server
      const completeRes = await fetchWithAuth('/api/creative/complete', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({ 
          jobId,
          assets: generatedImages
        })
      });

      if (!completeRes.ok) return;
      const completeData = await completeRes.json();
      if (completeData.error) throw new Error(completeData.error);

      // 5. Update UI
      setResult({ 
        description: productDescription || "Render complete. High-fidelity assets synthesized.", 
        images: generatedImages 
      });
      setJobStatus('done');
      setIsLoading(false);
      setActiveJobId(null);
    } catch (error: any) {
      console.error(error);
      setError(error.message || "Failed to connect to the rendering cluster. Please check your network connection.");
      setIsLoading(false);
      setJobStatus('failed');
    }
  };

  const handleCopyText = () => {
    if (result?.description) {
      navigator.clipboard.writeText(result.description);
      setIsCopied(true);
      setTimeout(() => setIsCopied(false), 2000);
    }
  };

  const handleRegenerateCopy = async () => {
    if (!productName || !result) return;
    setIsRegeneratingCopy(true);
    try {
      const service = new GeminiService();
      const newDescription = await service.generateProductDescription(productName);
      setResult({ ...result, description: newDescription });
    } catch (e) {
      console.error(e);
      setError("Failed to regenerate copy. Please try again.");
    } finally {
      setIsRegeneratingCopy(false);
    }
  };

  const handleDownloadAll = () => {
    if (!result) return;
    result.images.forEach((img, i) => {
      const variantName = i === 0 ? 'Studio' : i === 1 ? 'Editorial' : i === 2 ? 'Lifestyle' : 'Minimalist';
      const link = document.createElement('a');
      link.href = img;
      link.download = `AdStrat_${variantName}_${productName.replace(/\s+/g, '_')}.png`;
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
    });
  };

  const getPreviewPrompt = () => {
    const fidelity = `CRITICAL: The uploaded product photo shows the EXACT product. Maintain 100% fidelity to:
- Product shape and silhouette
- Exact colors (do not alter any color)
- All text, logos, labels, branding
- Product proportions and scale
Only change: lighting, background, environment.
The product itself must be pixel-perfect identical to the uploaded photo.`;

    const qualityText = quality === 'studio' ? "Ultra high resolution. Maximum detail. Print quality. 300 DPI equivalent." : "";

    const studioPrompt = `Professional commercial product photography. 
${productName}, ${productDetails}. 
Shot on Phase One IQ4 150MP medium format camera. 
Lens: Schneider Kreuznach 110mm f/2.8. 
Lighting: 3-point softbox setup — key light at 
45 degrees left, fill light at 1:3 ratio right, 
rim light behind for separation. 
Background: Pure white seamless paper #FFFFFF. 
Shadows: Soft natural drop shadow beneath product. 
Color grading: Neutral, accurate colors, 
slight warmth +200K. 
Style: Apple product photography, Zara e-commerce. 
Output: High resolution, razor sharp focus on product, 
zero background distractions. 
The product must be centered, fill 70% of frame. 
NO AI artifacts. NO distortion. 
Photorealistic — indistinguishable from real photography.`;

    return `${fidelity}\n\n${studioPrompt}\n\n${qualityText}`;
  };

  return (
    <div className="w-full h-full flex flex-col p-4 md:p-10 animate-fade overflow-y-auto scroll-hide bg-slate-50/30">
      <div className="max-w-5xl mx-auto w-full space-y-8">
        {/* Header Section */}
        <div className="bg-white border border-slate-200 p-8 md:p-12 rounded-[2.5rem] shadow-sm flex flex-col md:flex-row items-center justify-between gap-8">
          <div className="flex-1">
            <div className="flex items-center gap-3 mb-4">
              <span className="px-3 py-1 bg-indigo-50 text-indigo-600 rounded-full text-[10px] font-black uppercase tracking-widest border border-indigo-100">Asset Lab v4.0</span>
              <span className="px-3 py-1 bg-slate-100 text-slate-500 rounded-full text-[10px] font-black uppercase tracking-widest border border-slate-200">Pro Render Engine</span>
            </div>
            <h2 className="text-4xl md:text-5xl font-black text-slate-900 tracking-tight mb-4">Creative <span className="text-indigo-600">Studio.</span></h2>
            <p className="text-slate-500 font-medium max-w-md">Transform raw product data into high-converting visual assets using our neural rendering pipeline.</p>
            {hasKey === false && (
              <div className="mt-4 p-3 bg-amber-50 border border-amber-200 rounded-xl flex items-center gap-3 animate-pulse">
                <i className="fa-solid fa-key text-amber-600"></i>
                <p className="text-[10px] font-black text-amber-700 uppercase tracking-widest">API Key Required for 1K Renders</p>
                <button onClick={handleOpenKeySelector} className="ml-auto px-3 py-1 bg-amber-600 text-white rounded-lg text-[8px] font-black uppercase tracking-widest">Select Key</button>
              </div>
            )}
          </div>
          <div className="w-full md:w-auto flex flex-col items-center md:items-end gap-2">
             <div className="px-6 py-4 bg-slate-900 rounded-2xl text-white shadow-xl w-full md:w-48 text-center md:text-left">
                <p className="text-[9px] font-black text-slate-400 uppercase tracking-widest mb-1">Current Protocol</p>
                <p className="text-sm font-black tracking-tight">8K NEURAL RENDER</p>
             </div>
          </div>
        </div>
        
        {/* Configuration Grid */}
        <div className="grid grid-cols-1 lg:grid-cols-3 gap-8">
          <div className="lg:col-span-2 bg-white border border-slate-200 p-8 md:p-10 rounded-[2.5rem] shadow-sm space-y-8">
            <div className="grid grid-cols-1 md:grid-cols-2 gap-8">
              <div className="space-y-3">
                <label className="text-[10px] font-black uppercase text-slate-400 tracking-widest ml-1">Asset Identity</label>
                <div className="relative">
                  <i className="fa-solid fa-tag absolute left-5 top-1/2 -translate-y-1/2 text-slate-300 text-xs"></i>
                  <input 
                    type="text" 
                    value={productName}
                    onChange={(e) => setProductName(e.target.value)}
                    placeholder="e.g. Minimalist Leather Watch"
                    className="w-full bg-slate-50 border border-slate-200 rounded-2xl pl-12 pr-5 py-4 focus:outline-none focus:border-indigo-400 focus:bg-white transition-all font-bold text-slate-800 placeholder:text-slate-300"
                  />
                </div>
              </div>
              <div className="space-y-3">
                <label className="text-[10px] font-black uppercase text-slate-400 tracking-widest ml-1">Product Details</label>
                <div className="relative">
                  <i className="fa-solid fa-align-left absolute left-5 top-5 text-slate-300 text-xs"></i>
                  <textarea 
                    value={productDetails}
                    onChange={(e) => setProductDetails(e.target.value)}
                    placeholder="e.g. Brushed steel case, black leather strap, sapphire crystal"
                    className="w-full bg-slate-50 border border-slate-200 rounded-2xl pl-12 pr-5 py-4 focus:outline-none focus:border-indigo-400 focus:bg-white transition-all font-bold text-slate-800 placeholder:text-slate-300 h-[58px] resize-none"
                  />
                </div>
              </div>
            </div>

            <div className="grid grid-cols-1 md:grid-cols-2 gap-8">
              <div className="space-y-3">
                <label className="text-[10px] font-black uppercase text-slate-400 tracking-widest ml-1">Render Mode</label>
                <div className="flex gap-2">
                  <button 
                    onClick={() => setRenderMode('variant')}
                    className={`flex-1 py-4 rounded-2xl text-[10px] font-black uppercase tracking-widest transition-all ${
                      renderMode === 'variant' 
                        ? 'bg-indigo-600 text-white shadow-lg shadow-indigo-200' 
                        : 'bg-slate-50 border border-slate-200 text-slate-400 hover:bg-white'
                    }`}
                  >
                    Variant
                  </button>
                  <button 
                    onClick={() => setRenderMode('lifestyle')}
                    className={`flex-1 py-4 rounded-2xl text-[10px] font-black uppercase tracking-widest transition-all ${
                      renderMode === 'lifestyle' 
                        ? 'bg-indigo-600 text-white shadow-lg shadow-indigo-200' 
                        : 'bg-slate-50 border border-slate-200 text-slate-400 hover:bg-white'
                    }`}
                  >
                    Lifestyle
                  </button>
                </div>
              </div>
            </div>

            <div className="space-y-3">
              <label className="text-[10px] font-black uppercase text-slate-400 tracking-widest ml-1">Output Quality</label>
              <div className="flex gap-2 p-1 bg-slate-50 rounded-2xl border border-slate-200">
                <button 
                  onClick={() => setQuality('standard')}
                  className={`flex-1 py-3 rounded-xl text-[10px] font-black uppercase tracking-widest transition-all ${
                    quality === 'standard' 
                      ? 'bg-white text-slate-900 shadow-sm border border-slate-200' 
                      : 'text-slate-400 hover:text-slate-600'
                  }`}
                >
                  Standard (Faster)
                </button>
                <button 
                  onClick={() => setQuality('studio')}
                  className={`flex-1 py-3 rounded-xl text-[10px] font-black uppercase tracking-widest transition-all flex items-center justify-center gap-2 ${
                    quality === 'studio' 
                      ? 'bg-indigo-600 text-white shadow-lg shadow-indigo-200' 
                      : 'text-slate-400 hover:text-slate-600'
                  }`}
                >
                  <i className="fa-solid fa-crown text-[8px]"></i>
                  High Detail (Pro Key)
                </button>
              </div>
            </div>

            <div className="grid grid-cols-1 md:grid-cols-2 gap-8">
              <div className="space-y-3">
                <label className="text-[10px] font-black uppercase text-slate-400 tracking-widest ml-1">Product Reference</label>
                <div className="relative border-2 border-dashed border-slate-200 rounded-[2rem] h-48 flex items-center justify-center overflow-hidden group hover:border-indigo-300 transition-all bg-slate-50/50">
                  {image ? (
                    <div className="relative w-full h-full p-4">
                      <img src={image} className="h-full w-full object-contain rounded-xl z-10" />
                      <div className="absolute inset-0 bg-slate-900/40 opacity-0 group-hover:opacity-100 transition-opacity flex items-center justify-center backdrop-blur-sm">
                        <button onClick={() => setImage(null)} className="px-6 py-3 bg-rose-500 text-white rounded-xl font-black text-[10px] uppercase tracking-widest shadow-xl flex items-center gap-2">
                          <i className="fa-solid fa-trash-can"></i> Remove
                        </button>
                      </div>
                    </div>
                  ) : (
                    <div className="text-center group-hover:scale-105 transition-all p-4">
                      <div className="w-12 h-12 bg-white rounded-2xl flex items-center justify-center shadow-sm border border-slate-100 mx-auto mb-2">
                        <i className="fa-solid fa-camera text-indigo-500 text-lg"></i>
                      </div>
                      <p className="text-[10px] font-black uppercase text-slate-900 mb-1">Product Photo</p>
                      <input type="file" accept="image/*" onChange={(e) => handleImageUpload(e, 'product')} className="absolute inset-0 opacity-0 cursor-pointer z-10" />
                    </div>
                  )}
                </div>
              </div>

              <div className="space-y-3">
                <label className="text-[10px] font-black uppercase text-slate-400 tracking-widest ml-1">Style Inspiration (Optional)</label>
                <div className="relative border-2 border-dashed border-slate-200 rounded-[2rem] h-48 flex items-center justify-center overflow-hidden group hover:border-indigo-300 transition-all bg-slate-50/50">
                  {inspirationImage ? (
                    <div className="relative w-full h-full p-4">
                      <img src={inspirationImage} className="h-full w-full object-contain rounded-xl z-10" />
                      <div className="absolute inset-0 bg-slate-900/40 opacity-0 group-hover:opacity-100 transition-opacity flex items-center justify-center backdrop-blur-sm">
                        <button onClick={() => setInspirationImage(null)} className="px-6 py-3 bg-rose-500 text-white rounded-xl font-black text-[10px] uppercase tracking-widest shadow-xl flex items-center gap-2">
                          <i className="fa-solid fa-trash-can"></i> Remove
                        </button>
                      </div>
                    </div>
                  ) : (
                    <div className="text-center group-hover:scale-105 transition-all p-4">
                      <div className="w-12 h-12 bg-white rounded-2xl flex items-center justify-center shadow-sm border border-slate-100 mx-auto mb-2">
                        <i className="fa-solid fa-wand-magic-sparkles text-indigo-500 text-lg"></i>
                      </div>
                      <p className="text-[10px] font-black uppercase text-slate-900 mb-1">Style Reference</p>
                      <input type="file" accept="image/*" onChange={(e) => handleImageUpload(e, 'inspiration')} className="absolute inset-0 opacity-0 cursor-pointer z-10" />
                    </div>
                  )}
                </div>
              </div>
            </div>

            <button 
              onClick={handleProcess}
              disabled={isLoading || !image || !productName}
              className="w-full py-6 bg-indigo-600 text-white rounded-[1.5rem] font-black text-xs uppercase tracking-widest shadow-2xl shadow-indigo-200 disabled:opacity-20 hover:bg-indigo-700 transition-all active:scale-95 flex items-center justify-center gap-3"
            >
              {isLoading ? <i className="fa-solid fa-circle-notch fa-spin"></i> : <i className="fa-solid fa-sparkles"></i>}
              {isLoading ? `Synthesizing: ${jobStatus?.toUpperCase()}` : (inspirationImage ? 'Initiate Style Transfer' : 'Initiate Professional Render')}
            </button>

            {error && (
              <div className="p-6 bg-rose-50 border border-rose-100 rounded-2xl animate-in fade-in slide-in-from-top-4 duration-300">
                <div className="flex gap-4">
                  <div className="w-10 h-10 bg-rose-100 rounded-xl flex items-center justify-center flex-shrink-0">
                    <i className="fa-solid fa-triangle-exclamation text-rose-600"></i>
                  </div>
                  <div>
                    <h4 className="text-[10px] font-black uppercase text-rose-900 tracking-widest mb-1">Neural Engine Error</h4>
                    <p className="text-[11px] font-bold text-rose-600 leading-relaxed">{error}</p>
                    <button 
                      onClick={() => setError(null)}
                      className="mt-3 text-[9px] font-black uppercase text-rose-400 hover:text-rose-600 transition-colors tracking-widest"
                    >
                      Dismiss
                    </button>
                  </div>
                </div>
              </div>
            )}
          </div>

          <div className="space-y-8">
            {/* Status Panel */}
            <div className="bg-slate-900 rounded-[2.5rem] p-8 text-white shadow-2xl relative overflow-hidden">
              <div className="absolute top-0 right-0 p-6 opacity-10">
                <i className="fa-solid fa-microchip text-6xl"></i>
              </div>
              <h3 className="text-xs font-black uppercase tracking-[0.2em] text-slate-500 mb-6">Processing Status</h3>
              <div className="space-y-6">
                <div className="flex items-center justify-between">
                  <span className="text-[10px] font-black text-slate-400 uppercase">Engine Status</span>
                  <span className="flex items-center gap-2 text-[10px] font-black text-emerald-400 uppercase">
                    <div className="w-1.5 h-1.5 bg-emerald-500 rounded-full animate-pulse"></div>
                    Operational
                  </span>
                </div>
                <div className="flex items-center justify-between">
                  <span className="text-[10px] font-black text-slate-400 uppercase">Active Jobs</span>
                  <span className="text-[10px] font-black text-white uppercase">{activeJobId ? '1' : '0'}</span>
                </div>
                <div className="pt-4 border-t border-slate-800">
                  <p className="text-[9px] font-bold text-slate-500 italic leading-relaxed">"Neural weights optimized for high-contrast product photography."</p>
                </div>
              </div>
            </div>

            {/* Tips Panel */}
            <div className="bg-white border border-slate-200 rounded-[2.5rem] p-8 shadow-sm">
              <h3 className="text-[10px] font-black uppercase tracking-[0.2em] text-slate-400 mb-6">Pro Tips</h3>
              <div className="space-y-4">
                {[
                  { icon: 'fa-lightbulb', text: 'Use high-resolution source images for better texture mapping.' },
                  { icon: 'fa-wand-magic', text: 'Variant mode preserves geometry while testing new backgrounds.' },
                  { icon: 'fa-bolt', text: 'Lifestyle mode generates contextual environments for the product.' }
                ].map((tip, i) => (
                  <div key={i} className="flex gap-4">
                    <i className={`fa-solid ${tip.icon} text-indigo-500 mt-1 text-xs`}></i>
                    <p className="text-[11px] font-bold text-slate-600 leading-relaxed">{tip.text}</p>
                  </div>
                ))}
              </div>
            </div>
          </div>
        </div>

        {/* Results Section */}
        {result && (
          <div className="space-y-8 animate-fade pb-20">
             <div className="flex items-center gap-4">
                <div className="h-px bg-slate-200 flex-1"></div>
                <span className="text-[10px] font-black text-slate-400 uppercase tracking-[0.3em]">Synthesized Output</span>
                <div className="h-px bg-slate-200 flex-1"></div>
             </div>

             <div className="p-10 bg-white border border-slate-200 rounded-[3rem] shadow-sm">
                <div className="flex flex-col md:flex-row gap-10">
                   <div className="flex-1 space-y-8">
                      <div>
                        <p className="text-[10px] font-black text-indigo-600 uppercase tracking-widest mb-4">Neural Copy Synthesis</p>
                        <p className="text-2xl font-black text-slate-900 leading-tight italic">"{result.description}"</p>
                      </div>
                      <div className="flex gap-3">
                        <button 
                          onClick={handleCopyText}
                          className="px-6 py-3 bg-slate-900 text-white rounded-xl font-black text-[10px] uppercase tracking-widest hover:bg-black transition-all flex items-center gap-2"
                        >
                          {isCopied ? <i className="fa-solid fa-check text-emerald-400"></i> : <i className="fa-solid fa-copy"></i>}
                          {isCopied ? 'Copied' : 'Copy Text'}
                        </button>
                        <button 
                          onClick={handleRegenerateCopy}
                          disabled={isRegeneratingCopy}
                          className="px-6 py-3 bg-slate-100 text-slate-600 rounded-xl font-black text-[10px] uppercase tracking-widest hover:bg-slate-200 transition-all flex items-center gap-2 disabled:opacity-50"
                        >
                          {isRegeneratingCopy ? <i className="fa-solid fa-circle-notch fa-spin"></i> : <i className="fa-solid fa-rotate"></i>}
                          Regenerate Copy
                        </button>
                        <button 
                          onClick={handleDownloadAll}
                          className="px-6 py-3 bg-indigo-50 text-indigo-600 rounded-xl font-black text-[10px] uppercase tracking-widest hover:bg-indigo-100 transition-all flex items-center gap-2"
                        >
                          <i className="fa-solid fa-download"></i>
                          Download All Images
                        </button>
                      </div>
                   </div>
                   <div className="grid grid-cols-1 md:grid-cols-4 gap-4 w-full">
                      {result.images.map((img, i) => (
                        <div key={i} className="rounded-2xl overflow-hidden border border-slate-200 group relative shadow-md aspect-square bg-slate-50">
                           {img.startsWith('data:image') ? (
                             <>
                               <img src={img} className="w-full h-full object-cover transition-transform group-hover:scale-110" />
                               <div className="absolute inset-0 bg-slate-900/60 opacity-0 group-hover:opacity-100 transition-opacity flex items-center justify-center backdrop-blur-[2px]">
                                  <div className="flex gap-2">
                                    <a href={img} download={`render-${i}.png`} className="w-10 h-10 bg-white rounded-lg flex items-center justify-center text-slate-900 shadow-xl hover:scale-110 transition-all"><i className="fa-solid fa-download"></i></a>
                                    <button className="w-10 h-10 bg-indigo-600 rounded-lg flex items-center justify-center text-white shadow-xl hover:scale-110 transition-all"><i className="fa-solid fa-expand"></i></button>
                                  </div>
                               </div>
                             </>
                           ) : (
                             <div className="w-full h-full flex flex-col items-center justify-center p-6 text-center bg-slate-100">
                               <i className="fa-solid fa-triangle-exclamation text-amber-500 mb-3 text-xl"></i>
                               <p className="text-[9px] font-bold text-slate-600 leading-tight">
                                 The AI described this image instead of generating it. This happens when the model is under load. Try the Brief Only button — it gives you the professional photography direction to recreate this shot yourself or with a photographer.
                               </p>
                             </div>
                           )}
                           <div className="absolute bottom-3 left-3">
                             <span className="px-2 py-1 bg-white/90 backdrop-blur-sm rounded-md text-[8px] font-black uppercase text-slate-900 border border-slate-200">
                               {inspirationImage ? `Variant ${i+1}` : (i === 0 ? 'Studio Shot' : i === 1 ? 'Editorial' : i === 2 ? 'Lifestyle' : 'Minimalist')}
                             </span>
                           </div>
                        </div>
                      ))}
                   </div>
                </div>
             </div>
          </div>
        )}
      </div>
    </div>
  );
};

export default CreativeStudio;
