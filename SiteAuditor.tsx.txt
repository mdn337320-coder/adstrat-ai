import React, { useState, useEffect } from 'react';
import { GeminiService } from '../services/geminiService';
import { AuditResult } from '../types';

const SiteAuditor: React.FC = () => {
  const [url, setUrl] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const [loadingStep, setLoadingStep] = useState(0);
  const [result, setResult] = useState<AuditResult | null>(null);

  const steps = ["Search grounding...", "Conversion leaks...", "Competitor benchmarks...", "Synthesizing..."];

  useEffect(() => {
    let interval: any;
    if (isLoading) {
      interval = setInterval(() => setLoadingStep(prev => (prev + 1) % steps.length), 4000);
    }
    return () => clearInterval(interval);
  }, [isLoading]);

  const handleAudit = async () => {
    if (!url.trim()) return;
    setIsLoading(true);
    setResult(null);
    try {
      const service = new GeminiService();
      const data = await service.runWebsiteAudit(url);
      setResult(data);
    } catch (error: any) {
      console.error(error);
      if (error.message === "QUOTA_EXCEEDED") {
        alert("AI Quota exceeded. Please try again later.");
      } else {
        alert("Audit failed. The site might be blocking AI crawlers or the URL is invalid.");
      }
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <div className="w-full h-full flex flex-col p-4 md:p-10 animate-fade overflow-y-auto scroll-hide bg-slate-50/30">
      <div className="max-w-6xl mx-auto w-full space-y-8">
        {/* Header Section */}
        <div className="bg-white border border-slate-200 p-8 md:p-12 rounded-[2.5rem] shadow-sm flex flex-col lg:flex-row items-center justify-between gap-10">
          <div className="flex-1">
            <div className="flex items-center gap-3 mb-4">
              <span className="px-3 py-1 bg-emerald-50 text-emerald-600 rounded-full text-[10px] font-black uppercase tracking-widest border border-emerald-100">Conversion Node v4.0</span>
              <span className="px-3 py-1 bg-slate-100 text-slate-500 rounded-full text-[10px] font-black uppercase tracking-widest border border-slate-200">Deterministic Scan</span>
            </div>
            <h2 className="text-4xl md:text-5xl font-black text-slate-900 tracking-tight mb-4">Trust <span className="text-emerald-600">Auditor.</span></h2>
            <p className="text-slate-500 font-medium max-w-md">Identify conversion leaks and trust deficits across your landing pages with our multi-point neural audit.</p>
          </div>
          <div className="flex flex-col sm:flex-row gap-4 w-full lg:w-auto">
            <div className="relative flex-1 lg:w-80">
              <i className="fa-solid fa-link absolute left-5 top-1/2 -translate-y-1/2 text-slate-300 text-xs"></i>
              <input 
                type="text" 
                value={url} 
                onChange={(e) => setUrl(e.target.value)} 
                placeholder="e.g. yourbrand.com" 
                className="w-full bg-slate-50 border border-slate-200 rounded-2xl pl-12 pr-5 py-4 focus:outline-none focus:border-emerald-400 focus:bg-white transition-all font-bold text-slate-800 placeholder:text-slate-300" 
              />
            </div>
            <button 
              onClick={handleAudit} 
              disabled={isLoading || !url.trim()} 
              className="px-8 py-4 bg-emerald-600 text-white rounded-2xl font-black text-[10px] uppercase tracking-widest shadow-xl shadow-emerald-100 disabled:opacity-20 flex items-center justify-center gap-3 hover:bg-emerald-700 transition-all active:scale-95"
            >
              {isLoading ? <i className="fa-solid fa-satellite fa-spin"></i> : <i className="fa-solid fa-bolt-lightning"></i>}
              {isLoading ? 'Scanning' : 'Initiate Audit'}
            </button>
          </div>
        </div>

        {result ? (
          <div className="space-y-10 animate-fade pb-20">
            {/* Score Grid */}
            <div className="grid grid-cols-2 md:grid-cols-5 gap-6">
              {Object.entries(result.scores || {}).map(([key, score]: [string, any]) => (
                <div key={key} className="bg-white p-6 rounded-[2rem] border border-slate-200 text-center shadow-sm hover:shadow-md transition-all">
                  <p className="text-[9px] font-black text-slate-400 uppercase tracking-widest mb-3">{key.replace('_', ' ')}</p>
                  <div className="relative inline-block">
                    <svg className="w-16 h-16 transform -rotate-90">
                      <circle cx="32" cy="32" r="28" stroke="currentColor" strokeWidth="4" fill="transparent" className="text-slate-100" />
                      <circle cx="32" cy="32" r="28" stroke="currentColor" strokeWidth="4" fill="transparent" 
                        strokeDasharray={175.9} 
                        strokeDashoffset={175.9 - (175.9 * score) / 100}
                        className={`${score > 80 ? 'text-emerald-500' : score > 60 ? 'text-amber-500' : 'text-rose-500'} transition-all duration-1000`} 
                      />
                    </svg>
                    <span className="absolute inset-0 flex items-center justify-center text-lg font-black text-slate-900">{score}</span>
                  </div>
                </div>
              ))}
            </div>

            <div className="grid grid-cols-1 lg:grid-cols-3 gap-8">
              {/* Aggregate Score Card */}
              <div className="p-10 bg-slate-900 rounded-[3rem] text-center shadow-2xl relative overflow-hidden group">
                 <div className="absolute top-0 right-0 p-8 opacity-10 group-hover:scale-110 transition-transform">
                    <i className="fa-solid fa-award text-6xl text-white"></i>
                 </div>
                 <p className="text-[10px] font-black text-slate-500 uppercase tracking-[0.3em] mb-8">Aggregate Trust Score</p>
                 <div className="text-8xl font-black text-white mb-6 tracking-tighter">
                   {Math.round((Object.values(result.scores || {}) as number[]).reduce((a: number, b: number) => a + b, 0) / (Object.values(result.scores || {}).length || 1)) || '85'}
                   <span className="text-2xl text-slate-600 ml-1">/100</span>
                 </div>
                 <div className="px-6 py-3 bg-white/5 rounded-2xl border border-white/10 inline-block">
                    <p className="text-[10px] font-black text-emerald-400 uppercase tracking-widest">Neural Scan Verified</p>
                 </div>
                 <p className="text-xs font-bold text-slate-400 italic leading-relaxed mt-10 px-4">"Deterministic scan complete. AI insights synthesized against global benchmarks."</p>
              </div>

              {/* Performance Leaks */}
              <div className="lg:col-span-2 bg-white border border-slate-200 p-10 rounded-[3rem] shadow-sm">
                 <div className="flex items-center justify-between mb-10">
                    <h3 className="text-slate-900 font-black text-sm flex items-center gap-3 uppercase tracking-widest"><i className="fa-solid fa-shield-virus text-rose-500"></i> Conversion Leaks</h3>
                    <span className="text-[10px] font-black text-slate-400 uppercase tracking-widest">{(result.weaknesses || []).length} Issues Detected</span>
                 </div>
                 <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                    {(result.weaknesses || []).map((w, i) => (
                      <div key={i} className="p-6 bg-slate-50 border border-slate-100 rounded-2xl hover:bg-white hover:border-rose-100 hover:shadow-lg transition-all group">
                         <div className="flex justify-between items-center mb-4">
                            <span className="text-xs font-black text-slate-800 group-hover:text-rose-600 transition-colors">{w.title}</span>
                            <span className={`text-[8px] font-black px-2 py-1 rounded-full uppercase tracking-widest ${w.impact === 'High' ? 'bg-rose-500 text-white' : 'bg-slate-200 text-slate-600'}`}>{w.impact}</span>
                         </div>
                         <p className="text-xs text-slate-500 font-medium leading-relaxed">{w.issue}</p>
                      </div>
                    ))}
                 </div>
              </div>
            </div>

            <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
              {/* Strategic Alpha */}
              <div className="p-10 bg-indigo-600 rounded-[3rem] text-white shadow-xl shadow-indigo-100 relative overflow-hidden">
                 <div className="absolute -bottom-10 -right-10 opacity-10">
                    <i className="fa-solid fa-wand-magic-sparkles text-9xl"></i>
                 </div>
                 <h3 className="text-indigo-200 font-black text-[10px] uppercase tracking-[0.3em] mb-8 flex items-center gap-3"><i className="fa-solid fa-bolt"></i> Strategic Alpha</h3>
                 <div className="space-y-6">
                    {(result.suggestions || []).map((s, i) => (
                      <div key={i} className="flex gap-5 text-sm font-bold text-white/90 leading-relaxed group">
                         <span className="w-8 h-8 rounded-xl bg-white/10 text-white flex items-center justify-center flex-shrink-0 text-xs shadow-sm font-black border border-white/10 group-hover:bg-white group-hover:text-indigo-600 transition-all">{i+1}</span>
                         <p className="pt-1">{s}</p>
                      </div>
                    ))}
                 </div>
              </div>

              {/* Roadmap */}
              <div className="p-10 bg-white border border-slate-200 rounded-[3rem] shadow-sm">
                 <h3 className="text-slate-400 font-black text-[10px] uppercase tracking-[0.3em] mb-8 flex items-center gap-3"><i className="fa-solid fa-map-location-dot text-emerald-500"></i> Optimization Roadmap</h3>
                 <div className="space-y-6">
                    {(result.conversion_roadmap || []).map((r, i) => (
                      <div key={i} className="flex gap-6 group">
                         <div className="flex flex-col items-center">
                            <div className="w-4 h-4 rounded-full border-2 border-emerald-500 bg-white group-hover:bg-emerald-500 transition-colors"></div>
                            {i < (result.conversion_roadmap || []).length - 1 && <div className="w-0.5 h-full bg-slate-100 my-1"></div>}
                         </div>
                         <div className="pb-6">
                            <p className="text-[10px] font-black text-emerald-600 uppercase tracking-widest mb-1">{r.step}</p>
                            <p className="text-sm font-black text-slate-800 mb-1">{r.action}</p>
                            <p className="text-xs font-medium text-slate-400 italic">Expected: {r.expected_result}</p>
                         </div>
                      </div>
                    ))}
                 </div>
              </div>
            </div>

            {/* Grounding Sources */}
            {result.grounding_sources && result.grounding_sources.length > 0 && (
              <div className="pt-10 border-t border-slate-200">
                <div className="flex items-center gap-3 mb-6">
                  <i className="fa-solid fa-server text-slate-300"></i>
                  <h4 className="text-[10px] font-black text-slate-400 uppercase tracking-[0.3em]">Neural Grounding Sources</h4>
                </div>
                <div className="flex flex-wrap gap-3">
                  {result.grounding_sources.map((source, i) => (
                    <a key={i} href={source.uri} target="_blank" rel="noopener noreferrer" className="px-5 py-3 bg-white border border-slate-200 rounded-2xl text-[10px] font-bold text-slate-500 hover:text-indigo-600 hover:border-indigo-200 transition-all shadow-sm flex items-center gap-3">
                      <i className="fa-solid fa-link opacity-30"></i>
                      {source.title}
                    </a>
                  ))}
                </div>
              </div>
            )}
          </div>
        ) : (
          <div className="py-32 flex flex-col items-center justify-center text-center animate-fade">
             <div className="w-32 h-32 bg-slate-50 rounded-full flex items-center justify-center mb-8 shadow-inner border border-slate-100">
                <i className="fa-solid fa-radar text-slate-200 text-6xl"></i>
             </div>
             <h3 className="text-2xl font-black text-slate-900 mb-3">Initialize Conversion Audit</h3>
             <p className="text-slate-400 font-bold text-sm max-w-sm italic leading-relaxed">
                Enter your landing page URL to perform a multi-point scan for trust signals, conversion friction, and performance leaks.
             </p>
          </div>
        )}
      </div>
    </div>
  );
};

export default SiteAuditor;
